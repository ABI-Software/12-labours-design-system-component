<Story name="Checkbox Buttons From Array">{stories.Primary()}</Story>

#### Code

```js
data() {
  return {
    data: [
      {
        label: 1,
        display: "Option 1",
        disabled: false
      },
      {
        label: 2,
        display: "Option 2",
        disabled: false
      },
      {
        label: 3,
        display: "Option 3",
        disabled: false
      }
    ],
    checkboxVal: ''
  }
}
```

```html
<checkbox
  v-for="item in data"
  v-bind:key="item.label"
  v-model="checkboxVal"
  :label="item.label"
  :disabled="item.disabled"
  :display="item.display"
/>
```
